\begin{pergunta}
\perguntaa{}
\end{pergunta}
\begin{resposta}
A função triangular é definida como:
\begin{align*}
	\tri (t) = \begin{cases} 1 + t & -1 \le t \le 0 \\ 1 - t & 0 \le t \le 1 \\ 0 & |t| > 1 \end{cases}
\end{align*}
Pode-se calcular a transformada de Fourier diretamente através da definição:
\begin{align}
	\mathpzc{F} (\omega) & = \int_{-\infty}^{\infty} \tri (t) \; e^{- \jmath \omega t} \; dt \notag \\
	& = \int_{-1}^{0}  (1 + t) \; e^{- \jmath \omega t} \; dt + \int_{0}^{1} (1 - t) \; e^{- \jmath \omega t} \; dt \notag \\
	& = \int_{-1}^{1} e^{- \jmath \omega t} \; dt + \int_{-1}^{0} t e^{- \jmath \omega t} \; dt - \int_{0}^{1} t e^{- \jmath \omega t} \; dt \label{Eq1}
\end{align}
\hhlin
A integral indefinida
\begin{align*}
	\int t e^{at} \; dt 
\end{align*}
pode ser obtida através da técnica de integração por partes:
\begin{align*}
	\int u \; dv = u v - \int v \; du 
\end{align*}
fazendo-se
\begin{align*}
	\begin{matrix} u = t & \implies & du = dt \\ dv = e^{at} \; dt & & v = \frac{1}{a} e^{at} \end{matrix}
\end{align*}
Assim:
\begin{align}
	\int t \; e^{at} \; dt & = t \; \frac{1}{a} e^{at} - \int \frac{1}{a} e^{at} \; dt \notag \\
	& = \frac{1}{a} \; e^{at} \; \left( t - \frac{1}{a} \right) \label{Eq2}
\end{align}
\hhlin
Com base na equação \ref{Eq2}, a equação \ref{Eq1} se torna:
\begin{align}
	\mathpzc{F} (\omega) & = \left. \frac{1}{a} e^{at} \right|_{-1}^{1} + \left. \frac{1}{a} \; e^{at} \; \left( t - \frac{1}{a} \right) \right|_{-1}^{0} - \left. \frac{1}{a} \; e^{at} \; \left( t - \frac{1}{a} \right) \right|_{0}^{1} \notag \qquad \boxed{a = - \jmath \omega} \notag \\
	& = \frac{1}{a} \left( e^{a} - e^{-a} \right) + \frac{1}{a} \left( 1 \cdot \left[ 0 - \frac{1}{a} \right] - e^{-a} \; \left[-1 - \frac{1}{a} \right] \right) - \frac{1}{a} \left( e^{a} \; \left[ 1 - \frac{1}{a} \right] - 1 \cdot \left[ 0 - \frac{1}{a} \right] \right) \notag \\
	& = \frac{1}{a} \left( e^{a} - e^{-a} - \frac{1}{a} + e^{-a} \; \left[1 + \frac{1}{a} \right] - e^{a} \; \left[ 1 - \frac{1}{a} \right] - \frac{1}{a} \right) \notag \\
	& = \frac{1}{a^2} \left( -2 + e^{-a} + e^{a} \right) \notag \\
	& = \frac{1}{[- \jmath \omega]^2} \left( -2 + e^{-\jmath \omega} + e^{\jmath \omega} \right) \notag \\
	& = \frac{2}{\omega^2} \left( \frac{}{} 1 - \cos(\omega) \right) \label{Eq3}
\end{align}
\hlin
A transformada pode também ser calculada através da propriedade da convolução, reconhecendo-se que:
\begin{align*}
	\tri (t) = \rect(t) * \rect(t)
\end{align*}
onde $ \rect (t) $ é a função retangular, cuja transformada é conhecida:
\begin{align*}
	\Fou \{ \rect (t) \} & = \sinc \left( \frac{\omega}{2} \right) \\
	& = \frac{\sin \left( \frac{\omega}{2} \right)}{\frac{\omega}{2}} \\
	& = \frac{2}{\omega} \sin \left( \frac{\omega}{2} \right)
\end{align*}
Pela propriedade da convolução
\begin{align}
	\Fou \{ f(t) * g(t) \} = \Fou \{ f(t) \} \Fou \{ g(t) \} \qquad \implies \Fou \{ \tri (t) \} &= \Fou \{ \rect(t) \} \Fou \{ \rect(t) \} \notag \\
	& = \frac{2}{\omega} \sin \left( \frac{\omega}{2} \right) \cdot \frac{2}{\omega} \sin \left( \frac{\omega}{2}  \right) \notag \\
	& = \frac{4}{\omega^2} \sin ^2 \left( \frac{\omega}{2} \right) \label{Eq4}
\end{align}
Para demonstrar que as equações \ref{Eq3} e \ref{Eq4} são equivalentes, pode-se lançar mão da identidade $ \cos(2a) = 1 - 2 \sin^2(a) $ e escrever:
\begin{align*}
	\frac{2}{\omega^2} \left( \frac{}{} 1 - \cos(\omega) \right) & = \frac{2}{\omega^2} \left( 1 - \left[ 1 - 2 \sin ^2 \left(\frac{\omega}{2} \right) \right] \right) \\
	& = \frac{2}{\omega^2} \left( 2 \sin ^2 \left(\frac{\omega}{2} \right) \right) \\
	& = \frac{4}{\omega^2} \sin ^2 \left(\frac{\omega}{2} \right) \\
\end{align*}	
\end{resposta}
