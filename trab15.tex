\begin{pergunta}
\perguntaad{}
\end{pergunta}
\begin{resposta}
\begin{equation*}
	\begin{split}
	& s = \cfrac{1}{(1 - f) + \cfrac{f}{K}} \implies (1 - f) + \cfrac{f}{K} = \cfrac{1}{s} \implies ... \\
	& \qquad \implies \cfrac{f}{K} = \cfrac{1}{s} - (1 - f) \implies ... \\
	K & = \cfrac{f}{\cfrac{1}{s} + f - 1} \\
	& = \cfrac{0,44}{\cfrac{1}{1,17} + 0,44 - 1} \\
	& = 1.5 \\
	\end{split}
\end{equation*}
\end{resposta}
