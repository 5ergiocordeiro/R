	\end{itemize}

\section{LLM}
\subsection{O que é}
\begin{frame}
	\frametitle{LLLM}
	\begin{exampleblock}{}
		Monitoração do Limite de Carregamento de Linha (\emph{Line Loading Limit Monitoring})
	\end{exampleblock}		
	\begin{block}{}
		As linhas de transmissão frequentemente poderiam receber mais carga
	\end{block}
	\begin{itemize}
		\item A capacidade de carregamento é limitada pela temperatura
		\item As linhas geralmente são projetadas considerando-se critérios conservadores
		\item As linhas são operadas considerando-se limites teóricos
	\end{itemize}		
\end{frame}

\begin{frame}
	\frametitle{LLLM}
	\begin{block}{}
		Sincrofasores podem ser usados para se estimar o limite real de carregamento
	\end{block}
	\begin{itemize}
		\item Calcula-se a resistência real da linha
		\item Daí, a dissipação de calor e
		\item E então a temperatura corrente
	\end{itemize}
	\begin{block}{}
		Não é necessário instalar sensores de temperatura nas linhas. \\
		Ganho esperado: aumento da capacidade de despacho de potência para linhas onde o limite de carregamento teórico seja causa de congestões do sistema. \\
	\end{block}
\end{frame}

\subsection{Implementação}
\begin{frame}
	\frametitle{LLLM: dificuldades}
	\begin{itemize}
		\item A temperatura na verdade varia ao longo do comprimento da linha
		\item As linhas com frequência operam com desequilíbrio
		\item Pode haver derivações na linha
		\item Necessidade de coordenação com os relés de sobrecarga 
	\end{itemize}
\end{frame}
\begin{frame}
	\frametitle{LLLM: requisitos}
	\begin{itemize}
		\item Locação de uma PMU em cada lado da linha.
		\item Fasores de tensão de cada barramento e corrente de linha (sequência positiva ou trifásicos).
		\item Taxa mínima de aquisição de dados: uma amostra por segundo.
		\item Latência máxima: da ordem de segundos.
		\item Software necessário não é complexo
		\item Modelo térmico e climático complexo (mas que pode ir sendo aprimorado com o tempo)
	\end{itemize}
\end{frame}


\section{ERAC}
\subsection{O que é}
\begin{frame}
	\frametitle{ERAC}
	\begin{block}{}
		Os esquemas existentes não operam de forma otimizada
	\end{block}
	\begin{itemize}
		\item Os critérios operacionais são determinados a partir de estudos off-line, baseados no cenário de pior caso.
		\item Em geral, as condições reais são melhores que as estimadas,
		\item mas, em casos raros, podem ser piores.
	\end{itemize}		
\end{frame}
\begin{frame}
	\frametitle{ERAC}
	\begin{block}{}
		Sincrofasores podem ser usados para se estimar melhor as condições do sistema
	\end{block}
	\begin{itemize}
		\item determinar com maior precisão e antecedência se uma emergência acontecerá
		\item determinar se será suficiente adicionar geradores, desligar cargas ou desligar geradores
		\item determinar os geradores que se deve adicionar ou desligar, ou a quantidade exata de cargas a ser desligada
	\end{itemize}
\end{frame}

\subsection{Implementação}
\begin{frame}
	\frametitle{ERAC: requisitos}
	\begin{itemize}
		\item Locação de uma PMU em cada barramento onde forem requeridas medições e controles para detecção de oscilações (OLCA).
		\item Fasores de tensão de cada barramento e corrente de linha (sequência positiva).
		\item Taxa mínima de aquisição de dados: dez amostras por segundo.
		\item Latência máxima: 200 ms.
		\item Software: bastante complexo.
		\item Depende da preexistência das funções OLCA e EVSA.
	\end{itemize}
\end{frame}

\section{Identificação e classificação centralizada de faltas}
\subsection{O que é}

	\begin{block}{}
		A proteção convencional não é otimizada
	\end{block}

		\item Monitoram-se as condições locais.
		\item Relés atuam com base nas medições locais e alguma informação recebida de outros dispositivos.
		\item É muito difícil coordenar a proteção quando a rede é muito complexa.

\end{frame}
