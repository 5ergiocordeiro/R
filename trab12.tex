\section*{Parte A: Conceitos e Definições}
\begin{pergunta}
Definir/conceituar de forma fisicamente consistente:
\begin{listaef}[2]
	\item Grandeza física; grandeza física escalar e grandeza física vetorial;
	\item Campo; campo escalar e campo vetorial;
	\item Vetor unitário;
	\item Produto escalar entre dois vetores;
	\item Produto vetorial entre dois vetores;
	\item Gradiente de um campo escalar;
	\item Fluxo de um campo vetorial;
	\item Divergente de um campo vetorial;
	\item Teorema de Gauss (ou da Divergência);
	\item Rotacional de um campo vetorial;
	\item Teorema de Stokes;
	\item Campo vetorial conservativo;
	\item Campo vetorial não-conservativo;
	\item Laplaciano de um campo escalar;
	\item Laplaciano de um campo vetorial;
	\item Função Delta de Dirac unidimensional.
\end{listaef}
\end{pergunta}
\begin{resposta}
\begin{listaef}[2]
	\item \textbf{Grandeza física} é qualquer propriedade de um fenômeno físico. Ela é uma \textbf{grandeza física escalar} quando pode ser quantificada através de um valor numérico simples, acompanhado por uma unidade de medida; e uma \textbf{grandeza física vetorial} quando, para quantificá-la, se necessita especificar um vetor, ou seja, um conjunto de $ n $ valores, se considerado um espaço $ \mathbb{R}_n $.
	\item \textbf{Campo} é uma distribuição espacial de uma grandeza física. A distribuição de uma grandeza física escalar é um \textbf{campo escalar} e a distribuição de uma grandeza física vetorial é um \textbf{campo vetorial}.
	\item \textbf{Vetor unitário} é um vetor cujo módulo é igual à unidade.
	\item O \textbf{produto escalar entre dois vetores} é uma grandeza escalar que pode ser entendida geometricamente como o tamanho da projeção de um vetor sobre o outro.
	\item O \textbf{produto vetorial entre dois vetores} é uma grandeza vetorial que pode ser entendida geometricamente como a medida da área delimitada pelos vetores. A direção do resultado é perpendicular aos vetores e o sentido é dado, por convenção, pela regra da mão direita.
	\item O \textbf{gradiente de um campo escalar} $ f $ é um campo vetorial que corresponde à máxima variação de $ f $ em cada ponto do espaço.
	\item O \textbf{fluxo de um campo vetorial} $ \vec{v} $ através de uma superfície $ S $ é um valor escalar dado pela somatória, sobre todos os pontos de $ S $, da componente de $ \vec{v} $ perpendicular à superfície.
	\item O \textbf{divergente de um campo vetorial} $ \vec{v} $ é um campo escalar que mede o espalhamento de $ \vec{v} $ (ou seja, a componente radial de $ \vec{v} $) em cada ponto do espaço. É definido como a relação entre o fluxo de $ \vec{v} $ através de um volume infinitesimal $ v $ centrado no ponto e o tamanho de $ v $.
	\item O \textbf{Teorema de Gauss (ou da Divergência)} estabelece que o fluxo de um campo vetorial $ \vec{v} $ através de uma superfície \textbf{fechada} $ S $ é igual à somatória do divergente de $ \vec{v} $ em todos os pontos do espaço delimitado por $ S $.
	\item O \textbf{rotacional de um campo vetorial} $ \vec{v} $ e um campo vetorial que indica as componentes tangenciais de $ \vec{v} $ em cada ponto do espaço. É definido como a razão entre a máxima circulação de $ \vec{v} $ através de uma curva ao redor do ponto e a área delimitada por essa curva.
	\item O \textbf{Teorema de Stokes} estabelece que a \textbf{circulação} de um vetor $ \vec{v} $ sobre uma curva $ C $ é igual ao fluxo do rotacional de $ \vec{v} $ através da superfície delimitada por $ C $.
	\item Um campo é \textbf{conservativo}, ou \textit{\textbf{potencial}}, quando o trabalho realizado na translação, em equilíbrio, de um ponto a outro do espaço independe do caminho escolhido. Quando o campo é vetorial, ele deve ser \textit{\textbf{irrotacional}} nesse caso.
	\item Um campo é \textbf{não-conservativo} quando o trabalho realizado na translação, em equilíbrio, de um ponto a outro do espaço depende do caminho escolhido. Quando o campo é vetorial, ele deve ter rotacional não nulo.
	\item O \textbf{laplaciano de um campo escalar} $ f $ é a medida do espaçamento das superfícies $ f = \mathbf{constante} $ em cada ponto do espaço.
	\begin{inthisrev}
	Segundo \cite{MAXWELL 1873 1}:
		\begin{myquote}
		I propose therefore to call $ \lap{} q $ the \textit{concentration} of $ q $ at the point $ P $, because it indicates the excess of the value of $ q $ at that point over its mean in the neighbourhood of the point.
		\end{myquote} 
	\end{inthisrev}
	\item O \textbf{laplaciano de um campo vetorial} $ \vec{v} $ não tem interpretação física definida. Matematicamente, ele é a diferença entre o gradiente do divergente de $ \vec{v} $ e o rotacional do rotacional de $ \vec{v} $: \[ \qquad \vlap \vec{v} = \grad (\div \vec{v}) - \rot (\rot \vec{v}) \]
	\item A \textbf{função Delta de Dirac unidimensional} $ \delta(x - x_0) $ é uma \textit{função generalizada} (ou \textit{distribuição}) que possui valor nulo em todo o seu domínio, com exceção do ponto $ x = x_0 $, onde assume valor infinito. \\
\end{listaef}
\end{resposta}
\section*{Parte B: Aplicação de Conceitos e Definições}
\begin{pergunta}
1) Escreva a expressão cartesiana dos seguintes operadores diferenciais: a) $ f \grad{} $ ; b) $ \vec{A} \cdot \grad $ ; c) $ \vec{A} \times \grad $ .
\end{pergunta}
\begin{resposta}
a)
\begin{equation*}
	\begin{split}
	f \grad & = f \left( \ga{}{}{} \right) \\
	& =  f \px{} \a{x} + f \py{} \a{y} + f \pz{} \a{z}
    \end{split}
\end{equation*}
b)
\begin{equation*}
	\begin{split}
	\vec{A} \cdot \grad & = \left( \frac{}{} A_x \a{x} + A_y \a{y} + A_z \a{z} \right) \cdot \left( \ga{}{}{} \right) \\
    & = A_x \px{} \a{x} + A_y \py{} \a{y} + A_z \py{} \a{y} \\
    \end{split}
\end{equation*}
c)
\begin{equation*}
	\begin{split}
	\vec{A} \times \grad & = \left( \frac{}{} A_x \a{x} + A_y \a{y} + A_z \a{z} \right) \times \left( \ga{}{}{} \right) \\
    & = \left| \begin{matrix} \a{x} & \a{y} & \a{z}  \\ & & \\ A_x & A_y & A_z \\ & & \\ \px{} & \py{} & \pz{} \end{matrix} \right| \\
    \end{split}
\end{equation*}
\end{resposta}
\newpage
\begin{pergunta}
2) Determine: a) a expressão cartesiana do operador $ d \vec{l} \cdot \grad $ ; b) a relação entre o operador do item a) e a diferencial “d” de uma função dependente apenas do ponto.
\end{pergunta}
\begin{resposta}
a)
\begin{equation*}
	\begin{split}
	d \vec{l} \cdot \grad & = \left( \frac{}{} dx \; \a{x} + dy \; \a{y} + dz \; \a{z} \right) \cdot \left( \ga{}{}{} \right) \\
    & = \px{} dx + \py{} dy + \pz{} dz \\
    \end{split}
\end{equation*}
b)
\begin{equation*}
	\begin{split}
	d & = \px{} dx + \py{} dy + \pz{} dz + \frac{\partial}{\partial t} dt \\
	& = \px{} dx + \py{} dy + \pz{} dz + 0 \\
	& = d \vec{l} \cdot \grad \\
    \end{split}
\end{equation*}
\end{resposta}

\begin{pergunta}
3) Mostre que: a) $ d \vec{C} = ( d \vec{l} \cdot \grad ) \vec{C} + \frac{\partial \vec{C}}{\partial t} dt $ ; b) $ \frac{d \vec{C}}{dt} = ( \vec{V} \cdot \grad ) \vec{C} + \frac{\partial \vec{C}}{\partial t} $, onde $ \vec{C} = \frac{d \vec{l}}{dt} $ .
\end{pergunta}
\begin{resposta}
a)
\begin{equation*}
	\begin{split}
	d \vec{C} & = \px{\vec{C}} dx + \py{\vec{C}} dy + \pz{\vec{C}} dz + \frac{\partial \vec{C}}{\partial t} dt \\
	& = \left( \px{} dx + \py{} dy + \pz{} dz \right) \vec{C} + \frac{\partial \vec{C}}{\partial t} dt \\
    & = \left[ \left( \frac{}{} dx \; \a{x} + dy \; \a{y} + dz \; \a{z} \right) \cdot \left( \ga{}{}{} \right) \right] \vec{C} + \frac{\partial \vec{C}}{\partial t} dt \\ 
	& = ( d \vec{l} \cdot \grad ) \vec{C} + \frac{\partial \vec{C}}{\partial t} dt \\		
    \end{split}
\end{equation*}
b)
\begin{equation*}
	\begin{split}
	\frac{d \vec{C}}{dt} & = \frac{1}{dt} \; d \vec{C} \\
	& = \frac{1}{dt} \left[ \left( \px{} dx + \py{} dy + \pz{} dz \right) \vec{C} + \frac{\partial \vec{C}}{\partial t} dt \right] \\
	& = \left( \px{} \frac{dx}{dt} + \py{} \frac{dy}{dt} + \pz{} \frac{dz}{dt} \right) \vec{C} + \frac{\partial \vec{C}}{\partial t} \\	
	& = \left[ \left( \frac{dx}{dt} \a{x} + \frac{dy}{dt} \a{y} + \frac{dz}{dt} \a{z} \right) \cdot \left( \ga{}{}{} \right) \right] \vec{C} + \frac{\partial \vec{C}}{\partial t} \\
	& = \left[ \vec{V} \cdot \grad \right] \vec{C} + \frac{\partial \vec{C}}{\partial t} \\
    \end{split}
\end{equation*}
\end{resposta}

\begin{pergunta}
4) Usando coordenadas e componentes cartesianos, demostre que: a) $ ( \vec{A} \cdot \grad ) \vec{r} = \\
\vec{A} $ ; b) $ ( \vec{A} \times \grad ) \cdot \vec{r} = 0 $ ; c) $ ( \vec{A} \times \grad ) \times \vec{r} = - 2 \vec{A} $ .
\end{pergunta}
\begin{resposta}
a)
\begin{equation*}
	\begin{split}
	( \vec{A} \cdot \grad ) \vec{r} & = \left[ \left( \frac{}{} A_x \a{x} + A_y \a{y} + A_z \a{z} \right) \cdot \left( \ga{}{}{} \right) \right] ( x \a{x} + ... \\
	& \qquad + y \a{y} + z \a{z} ) \\
	& = \left[ A_x \px{} + A_y \py{} + A_z \pz{} \right] ( x \a{x} + y \a{y} + z \a{z} ) \\
	& = A_x \px{} ( x \a{x} + y \a{y} + z \a{z} ) + A_y \py{} ( x \a{x} + y \a{y} + z \a{z} ) + ... \\
	& \qquad + A_z \pz{} ( x \a{x} + y \a{y} + z \a{z} ) \\
	& = A_x \a{x} + \vec{0} + \vec{0} + \vec{0} + A_y \a{y} + \vec{0} + \vec{0} + \vec{0} + A_z \a{z} \\
	& = \vec{A} \\
    \end{split}
\end{equation*}
\newpage
b)
\begin{equation*}
	\begin{split}
	( \vec{A} \times \grad ) \cdot \vec{r} & = \left| \begin{matrix} \a{x} & \a{y} & \a{z}  \\ & & \\ A_x & A_y & A_z \\ & & \\ \px{} & \py{} & \pz{} \end{matrix} \right| \cdot ( x \a{x} + y \a{y} + z \a{z} ) \\
	& = \left[ \left( A_y \pz{} - A_z \py{} \right) \a{x} + \left( A_z \px{} - A_x \pz{} \right) \a{y} + \right. ... \\
	& \qquad + \left. \left( A_x \py{} - A_y \px{} \right) \a{z} \right] \cdot \cdot ( x \a{x} + y \a{y} + z \a{z} ) \\
	& = \left( A_y \pz{} - A_z \py{} \right) x + \left( A_z \px{} - A_x \pz{} \right) y + ... \\
	& \qquad + \left( A_x \py{} - A_y \px{} \right) z \\
	& = 0 + 0 + 0 \\  
	& = 0 \\
    \end{split}
\end{equation*}
\newpage
c)
\begin{equation*}
	\begin{split}
	( \vec{A} \times \grad ) \times \vec{r} & = \left| \begin{matrix} \a{x} & & \a{y} & & \a{z}  \\ & & & & \\ A_y \pz{} - A_z \py{} & & A_z \px{} - A_x \pz{} & & A_x \py{} - A_y \px{} \\ & & & & \\ x & & y & & z \end{matrix} \right| \\ \\
	& = \left[ \left( A_z \px{} - A_x \pz{} \right) z - \left( A_x \py{} - A_y \px{} \right) y \right] \hat{a_x} + ... \\
	& \qquad + \left[ \left( A_x \py{} - A_y \px{} \right) x - \left( A_y \pz{} - A_z \py{} \right) z \right] \a{y} + ... \\
	& \qquad + \left[ \left( A_y \pz{} - A_z \py{} \right) y - \left( A_z \px{} - A_x \pz{} \right) x \right] \a{z} \\ \\
	& = \left[ - A_x - A_x \right] \hat{a_x} + \left[ - A_y - A_y \right] \a{y} + \left[ - A_z - A_z \right] \a{z} \\
	& = - 2 \vec{A}
    \end{split}
\end{equation*}
\end{resposta}

\begin{pergunta}
5) Usando coordenadas e componentes cartesianos, demonstre que $ ( \vec{A} \cdot \grad) \frac{\vec{r}}{r^n} = \frac{\vec{A} - n(\vec{A} \cdot \a{r}) \a{r}}{r^n} $ .
\end{pergunta}
\begin{resposta}
\begin{equation*}
	\begin{split}
	( \vec{A} \cdot \grad ) \frac{\vec{r}}{r^n} & = ( \vec{A} \cdot \grad ) ( \vec{r} \; r^{-n} ) \\
	& = \left[ \frac{}{} ( \vec{A} \cdot \grad ) \vec{r} \right] r^{-n} + \left[ ( \vec{A} \cdot \grad ) r^{-n} \right] \vec{r} \\
	& = \vec{A} \; r^{-n} + \left[ \left( A_x \px{} + A_y \py{} + A_z \pz{} \right) (x^2 + y^2 + z^2)^\frac{-n}{2} \right] \vec{r} \\
	& = \frac{\vec{A}}{r^n} + \left[ A_x \left( \frac{-n}{2} \right) (x^2 + y^2 + z^2)^\frac{-n-2}{2} \; (2x) + \right. ... \\
	& \qquad A_y \left( \frac{-n}{2} \right) (x^2 + y^2 + z^2)^\frac{-n-2}{2} \; (2y) + ... \\
	& \qquad + \left. A_z \left( \frac{-n}{2} \right) (x^2 + y^2 + z^2)^\frac{-n-2}{2} \; (2z) \right] \vec{r} \\
	& = \frac{\vec{A}}{r^n} - n \; \frac{x \; A_x + y \; A_y + z \; A_z}{(x^2 + y^2 + z^2)^\frac{n+2}{2}} \; \vec{r} \\
	& = \frac{\vec{A}}{r^n} - n \; \frac{\vec{A} \cdot \vec{r}}{r^{n+2}} \; \vec{r} \\
	& = \frac{\vec{A}}{r^n} - n \; \frac{\vec{A} \cdot \a{r}}{r^n} \; \a{r} \\
    \end{split}
\end{equation*}
\end{resposta}

\begin{pergunta}
6) Na mecânica quântica, define-se o operador vetorial momento angular como $ \vec{L} = -j \vec{r} \times \grad $ , onde $ j \equiv \sqrt{-1} $ . Mostre que $ \vec{L} \times \vec{L} = j \vec{L} $ .
\end{pergunta}
\begin{resposta}
\begin{equation*}
	\begin{split}
	\vec{L} \times \vec{L} & = ( -j \vec{r} \times \grad ) \times ( -j \vec{r} \times \grad ) \\
	& = -j \left| \begin{matrix} \a{x} & \a{y} & \a{z}  \\ & & \\ x & y & z \\ & & \\ \px{} & \py{} & \pz{} \end{matrix} \right| \times -j \left| \begin{matrix} \a{x} & \a{y} & \a{z}  \\ & & \\ x & y & z \\ & & \\ \px{} & \py{} & \pz{} \end{matrix} \right| \\
	& = j^2 \left[ \left( y \pz{} - z \py{} \right) \a{x} + \left( z \px{} - x \pz{} \right) \a{y} + \left( x \py{} - y \px{} \right) \a{z} \right] \times ... \\
	& \qquad \times \left[ \left( y \pz{} - z \py{} \right) \a{x} + \left( z \px{} - x \pz{} \right) \a{y} + \left( x \py{} - y \px{} \right) \a{z} \right] \\
	& \\
	& = - \left| \begin{matrix} \a{x} & & \a{y} & & \a{z}  \\ & & \\ y \pz{} - z \py{} & & z \px{} - x \pz{} & & x \py{} - y \px{} \\ & & \\ y \pz{} - z \py{} & & z \px{} - x \pz{} & & x \py{} - y \px{} \end{matrix} \right| \\
	& \\
	& = - \left\{ \left[ \left(z \px{} - x \pz{} \right) \left( x \py{} - y \px{} \right) - \left( x \py{} - y \px{} \right) \left(z \px{} - x \pz{} \right) \right] \a{x} + \right. ... \\
	& \qquad + \left. \left[ \left(x \py{} - y \px{} \right) \left(y \pz{} - z \py{} \right) - \left(y \pz{} - z \py{} \right) \left(x \py{} - y \px{} \right) \right] \a{y} \right\} + ... \\
	& \qquad + \left. \left[ \left(y \pz{} - z \py{} \right) \left(z \px{} - x \pz{} \right) - \left(z \px{} - x \pz{} \right) \left(y \pz{} - z \py{} \right) \right] \a{z} \right\} \\
	& \qquad \qquad \vdots \\
	\end{split}
\end{equation*}
\newpage
\begin{equation*}

\begin{pergunta}
11) Mostre que $ \vec{t} = \grad f \times \grad g $ é um vetor tangente à curva determinada pela interseção das superfícies $ f{(\vec{r})}  = \mathbf{constante} $ e $ g{(\vec{r})} = \mathbf{constante} $ .
\end{pergunta}
\begin{resposta}
De acordo com as propriedades do gradiente, $ \grad f $ será um vetor ortogonal à superfície $ f{(\vec{r})} = \mathbf{constante} $ e $ \grad g $ será um vetor ortogonal à superfície $ g{(\vec{r})} = \mathbf{constante} $. Todos os pontos da curva $ C $, determinada pela interseção dessas superfícies, pertencem a ambas simultaneamente. Portanto, em cada um desses pontos  os vetores $ \grad f $ e $ \grad g $ serão ortogonais a $ C $. O produto vetorial de dois vetores é sempre ortogonal a ambos os vetores, portanto, $ \vec{t} $ estará na mesma direção da curva $ C $ em cada um dos seus pontos. \\
\textbf{Isso não é válido no caso especial de os vetores $ \grad f $ e $ \grad g $ estarem na mesma direção (por exemplo, duas superfícies cilíndricas tangentes na direção dos eixos de simetria), porque nesse caso $ \vec{t} $ será nulo.}
\end{resposta}

	& = \iint \limits_S ( C_n (n+1) \rho^{n-1}] ) (\rho d \rho d \phi)  \\

\begin{pergunta}
46) Partindo da equação $ \delta(\vec{r} - \vec{r}_0) = \delta(x - x_0) \delta(y - y_0) \delta(z - z_0) $ , deduza as seguintes equações: a) $ \delta(\vec{r} - \vec{r}_0) = 0 $ , se $ \vec{r} \ne \vec{r}_0 $ ; b) $ \iiint \limits_v \delta(\vec{r} - \vec{r}_0) dv = \begin{cases} 0, \; se \; \vec{r}_0 \not \in v \\ 1, \; se \; \vec{r}_0 \in v \end{cases} $ ; \\
c) $ \iiint \limits_v f(\vec{r}) \delta(\vec{r} - \vec{r}_0) dv = \begin{cases} 0, \; se \; \vec{r}_0 \not \in v \\ f(\vec{r}_0), \; se \; \vec{r}_0 \in v \end{cases} $ .
\end{pergunta}
\begin{resposta}
a)
\begin{equation*}
	\begin{split}
	& \vec{r} \ne \vec{r}_0 \implies x \ne x_0 \lor y \ne y_0 \lor z \ne z_0 \\
	& \qquad \delta(x - x_0) = 0 \lor \delta(y - y_0) = 0 \lor \delta(z - z_0) = 0 \\
	& \qquad \delta(x - x_0) \delta(y - y_0) \delta(z - z_0) = 0 \\
	& \qquad \delta(\vec{r} - \vec{r}_0) = 0 \\
	\end{split}
\end{equation*}
b)
\begin{equation*}
	\begin{split}
	& r_0 \not \in v \implies \delta(\vec{r} - \vec{r}_0) = 0 \forall \vec{r} \in v \\
	\qquad \iiint \limits_v \delta(\vec{r} - \vec{r}_0) dv & = \iiint \limits_v 0 \; dv \\
	\qquad & = 0 \\
	\end{split}
\end{equation*}
\begin{equation*}
	\begin{split}
	& r_0 \in v \implies \delta(\vec{r} - \vec{r}_0) = \begin{cases} 0, & \vec{r} \ne \vec{r}_0 \\ \delta(0), & \vec{r} = \vec{r}_0 \end{cases} \\
	\qquad \iiint \limits_v \delta(\vec{r} - \vec{r}_0) dv & = \iiint \limits_v \delta(0) \; dv \\
	& = 1 \\
	\end{split}
\end{equation*}

c)
\begin{equation*}
	\begin{split}
	& r_0 \not \in v \implies \delta(\vec{r} - \vec{r}_0) = 0 \forall \vec{r} \in v \\
	\qquad \iiint \limits_v f(\vec{r}) \delta(\vec{r} - \vec{r}_0) dv & = \iiint \limits_v 0 \; dv \\
	\qquad & = 0 \\

\end{equation*}

\begin{pergunta}
47) Mostre que: $ \delta(x) = \lim \limits_{k \to \infty} \left( \frac{k}{\sqrt{\pi}} e^{-k^2x^2} \right) $ .
\end{pergunta}
\begin{resposta}
\begin{equation*}
	\begin{split}
	 \left. \lim \limits_{k \to \infty} \left( \frac{k}{\sqrt{\pi}} e^{-k^2x^2} \right) \right|_{x = 0} & = \lim \limits_{k \to \infty} \left( \frac{k}{\sqrt{\pi}} e^0 \right) \\
	 & = \infty \\
	 \left. \lim \limits_{k \to \infty} \left( \frac{k}{\sqrt{\pi}} e^{-k^2x^2} \right) \right|_{x \ne 0} & = \lim \limits_{k \to \infty} \left( \frac{k}{\sqrt{\pi}} e^{-k2} \right) \\
	 & = 0 \\
	\end{split}
\end{equation*}
\begin{equation*}
	\begin{split}
	\int_{-\infty}^{\infty} \lim \limits_{k \to \infty} \left( \frac{k}{\sqrt{\pi}} e^{-k^2x^2} \right) dx & = \lim \limits_{k \to \infty} \left( \int_{-\infty}^{\infty} \frac{k}{\sqrt{\pi}} e^{-k^2x^2} dx \right) \\
	& = \lim \limits_{\sigma \to 0} \left( \int_{-\infty}^{\infty} \frac{1}{\sigma \sqrt{2 \pi}} e^{-\frac{x^2}{2 \sigma^2}} dx \right) \\
	& = \lim \limits_{\sigma \to 0} 1 \\
	& = 1 \\
	\end{split}
\end{equation*}
\end{resposta}

49) Mostre que: $ \int_{-\beta}^{\beta} e^{jkx} dk = \frac{2}{x} \sin \beta x $ .
\end{pergunta}
\begin{resposta}
\begin{equation*}
	\begin{split}
	\int_{-\beta}^{\beta} e^{jkx} dk & = \left. \frac{1}{jx} \; e^{jkx} \right|_{-\beta}^{\beta} \\
	& = \frac{e^{j \beta x} - e^{- j \beta x}}{jx} \\
	& = \frac{2}{x} \; \frac{e^{j \beta x} - e^{- j \beta x}}{j} \\
	& = \frac{2}{x} \sin \beta x \\
	\end{split}
\end{equation*}
\end{resposta}

\begin{pergunta}
50) Mostre que: $ \delta(x) = \frac{1}{2 \pi} \int_{-\infty}^{\infty} e^{jkx} dk $ .
\end{pergunta}
\begin{resposta}
\begin{equation*}
	\begin{split}
	x = 0 \implies \frac{1}{2 \pi} \int_{-\infty}^{\infty} e^{jkx} dk & = \frac{1}{2 \pi} \left( \int_{-\infty}^{\infty} e^{0} dk \right) \\
	& = \frac{1}{2 \pi} \int_{-\infty}^{\infty} dk \\
	& = \infty \\
	x \ne 0 \implies \frac{1}{2 \pi} \int_{-\infty}^{\infty} e^{jkx} dk & = \frac{1}{2 \pi} \lim \limits_{\beta \to \infty} \int_{-\beta}^{\beta} e^{jkx} dk \\
	& = \frac{1}{2 \pi} \lim \limits_{\beta \to \infty} \left( \frac{2}{x} \sin \beta x \right) \\
	& = \frac{1}{\pi x} \lim \limits_{\beta \to \infty} (\sin \beta x) \\
	& = \mathbf{indefinido} \\
	\end{split}
\end{equation*}
\\
\begin{equation*}
	\begin{split}
	\int_{-\infty}^{\infty} \left( \frac{1}{2 \pi} \int_{-\infty}^{\infty} e^{jkx} dk  \right) dx & = \lim \limits_{\beta \to \infty} \left[ \int_{-\infty}^{\infty} \left( \frac{1}{\pi x} \sin \beta x \right) dx \right] \\
	& = \frac{1}{\pi} \lim \limits_{\beta \to \infty} \left[ \int_{-\infty}^{\infty} \frac{\sin \beta x}{\beta x} \beta dx \right] \\
	& = \frac{1}{\pi} \lim \limits_{\beta \to \infty} \left[ \int_{-\infty}^{\infty} \sinc(u) \; du \right] \\
	& = \frac{1}{\pi} \lim \limits_{\beta \to \infty} \pi \\
	& = 1 \\
	\end{split}
\end{equation*}

Apesar de o valor da função ser indefinido para $ x \ne 0 $, ela exibe a propriedade importante, em termos práticos, que é a da filtragem (\textit{sifting}), portanto pode ser considerada como uma expressão da Delta de Dirac.
\end{resposta}
